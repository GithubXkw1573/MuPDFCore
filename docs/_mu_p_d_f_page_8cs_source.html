24  /// <summary>
25  /// A wrapper over a MuPDF page object, which contains information about the page's boundaries.
26  /// </summary>
27  public class MuPDFPage : IDisposable
28  {
29  /// <summary>
30  /// The page's bounds at 72 DPI. Read-only.
31  /// </summary>

39  /// <summary>
40  /// A pointer to the native page object.
41  /// </summary>
42  internal readonly IntPtr NativePage;
43 
44  /// <summary>
45  /// The context that owns the document from which this page was extracted.
46  /// </summary>
47  private readonly MuPDFContext OwnerContext;
48 
49  /// <summary>
50  /// The document from which the page was extracted.
51  /// </summary>
52  internal readonly MuPDFDocument OwnerDocument;
53 
54  /// <summary>
55  /// The page's original bounds. Read-only.
56  /// </summary>
57  internal Rectangle OriginalBounds { get; }
58 
59  /// <summary>
60  /// Create a new <see cref="MuPDFPage"/> object from the specified document.
61  /// </summary>
62  /// <param name="context">The context that owns the document.</param>
63  /// <param name="document">The document from which the page should be extracted.</param>
64  /// <param name="number">The number of the page that should be extracted (starting at 0).</param>
65  internal MuPDFPage(MuPDFContext context, MuPDFDocument document, int number)
66  {
67  this.OwnerContext = context;
68  this.OwnerDocument = document;
69  this.PageNumber = number;
70 
71  float x = 0;
72  float y = 0;
73  float w = 0;
74  float h = 0;
75 
76  ExitCodes result = (ExitCodes)NativeMethods.LoadPage(context.NativeContext, document.NativeDocument, number, ref NativePage, ref x, ref y, ref w, ref h);
77 
78  this.Bounds = new Rectangle(Math.Round(x * document.ImageXRes / 72.0 * 1000) / 1000, Math.Round(y * document.ImageYRes / 72.0 * 1000) / 1000, Math.Round(w * document.ImageXRes / 72.0 * 1000) / 1000, Math.Round(h * document.ImageYRes / 72.0 * 1000) / 1000);
79  this.OriginalBounds = new Rectangle(x, y, w, h);
80 
81  switch (result)
82  {
83  case ExitCodes.EXIT_SUCCESS:
84  break;
85  case ExitCodes.ERR_CANNOT_LOAD_PAGE:
86  throw new MuPDFException("Cannot load page", result);
87  case ExitCodes.ERR_CANNOT_COMPUTE_BOUNDS:
88  throw new MuPDFException("Cannot compute bounds", result);
89  default:
90  throw new MuPDFException("Unknown error", result);
91  }
92  }
93 
94  private bool disposedValue;
95 
96  ///<inheritdoc/>
97  protected virtual void Dispose(bool disposing)
98  {
99  if (!disposedValue)
100  {
101  NativeMethods.DisposePage(OwnerContext.NativeContext, NativePage);
102  disposedValue = true;
103  }
104  }
105 
106  ///<inheritdoc/>
107  ~MuPDFPage()

109  Dispose(disposing: false);
110  }
111 
112  ///<inheritdoc/>
113  public void Dispose()
114  {
115  Dispose(disposing: true);
116  GC.SuppressFinalize(this);
117  }
118  }

120  /// <summary>
121  /// A lazy collection of <see cref="MuPDFPage"/>s. Each page is loaded from the document as it is requested for the first time.
122  /// </summary>
123  public class MuPDFPageCollection : IReadOnlyList<MuPDFPage>, IDisposable
124  {
125  /// <summary>
126  /// The internal store of the pages.
127  /// </summary>
128  private readonly MuPDFPage[] Pages;
129 
130  /// <summary>
131  /// The context that owns the document from which the pages were extracted.
132  /// </summary>
133  private readonly MuPDFContext OwnerContext;
134 
135  /// <summary>
136  /// The document from which the pages were extracted.
137  /// </summary>
138  private readonly MuPDFDocument OwnerDocument;
139 
140  /// <summary>
141  /// The number of pages in the collection.
142  /// </summary>
143  public int Length { get { return Pages.Length; } }
144 
145  /// <summary>
146  /// The number of pages in the collection.
147  /// </summary>
148  public int Count { get { return Pages.Length; } }
149 
150  /// <summary>
151  /// Get a page from the collection.
152  /// </summary>
153  /// <param name="index">The number of the page (starting at 0).</param>
154  /// <returns>The specified <see cref="MuPDFPage"/>.</returns>
155  public MuPDFPage this[int index]
156  {
157  get
158  {
159  if (index < 0 || index > Pages.Length)
160  {
161  throw new IndexOutOfRangeException();
162  }
163 
164  if (Pages[index] == null)
165  {
166  Pages[index] = new MuPDFPage(OwnerContext, OwnerDocument, index);
167  }
168 
169  return Pages[index];
170  }
171  }
172 
173  /// <summary>
174  /// Create a new <see cref="MuPDFPageCollection"/> from the specified document, containing the specified number of pages.
175  /// </summary>
176  /// <param name="context">The context that owns the document.</param>
177  /// <param name="document">The document from which the pages should be extracted.</param>
178  /// <param name="length">The number of pages in the document.</param>
179  internal MuPDFPageCollection(MuPDFContext context, MuPDFDocument document, int length)
180  {
181  Pages = new MuPDFPage[length];
182  OwnerContext = context;
183  OwnerDocument = document;
184  }
185 
186  ///<inheritdoc/>
187  public IEnumerator<MuPDFPage> GetEnumerator()
188  {
189  for (int i = 0; i < Pages.Length; i++)
190  {
191  if (Pages[i] == null)
192  {
193  Pages[i] = new MuPDFPage(OwnerContext, OwnerDocument, i);
194  }
195  }
196 
197  return ((IEnumerable<MuPDFPage>)Pages).GetEnumerator();
198  }
199 
200  ///<inheritdoc/>
201  IEnumerator IEnumerable.GetEnumerator()
202  {
203  return GetEnumerator();
204  }
205 
206  private bool disposedValue;
207 
208  ///<inheritdoc/>
209  protected virtual void Dispose(bool disposing)
210  {
211  if (!disposedValue)
212  {
213  if (disposing)
214  {
215  for (int i = 0; i < Pages.Length; i++)
216  {
217  Pages[i]?.Dispose();
218  }
219  }
220 
221  disposedValue = true;
222  }
223  }
224 
225  ///<inheritdoc/>
226  public void Dispose()
227  {
228  Dispose(disposing: true);
229  GC.SuppressFinalize(this);
230  }
231  }
