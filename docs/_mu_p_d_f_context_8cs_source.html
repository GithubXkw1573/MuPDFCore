22 /// <summary>
23 /// A wrapper around a MuPDF context object, which contains the exception stack and the resource cache store.
24 /// </summary>
25 public class MuPDFContext : IDisposable
26 {
27 /// <summary>
28 /// A pointer to the native context object.
29 /// </summary>
30 internal readonly IntPtr NativeContext;
31
32 /// <summary>
33 /// The current size in bytes of the resource cache store. Read-only.
34 /// </summary>
35 public long StoreSize
36 {
37 get
38 {
39 return (long)NativeMethods.GetCurrentStoreSize(this.NativeContext);
40 }
41 }
42
43 /// <summary>
44 /// The maximum size in bytes of the resource cache store. Read-only.
45 /// </summary>
46 public long StoreMaxSize
47 {
48 get
49 {
50 return (long)NativeMethods.GetMaxStoreSize(this.NativeContext);
51 }
52 }
53
54 /// <summary>
55 /// Create a new <see cref="MuPDFContext"/> instance with the specified cache store size.
56 /// </summary>
57 /// <param name="storeSize">The maximum size in bytes of the resource cache store. The default value is 256 MiB.</param>
58 public MuPDFContext(uint storeSize = 256 << 20)
59 {
60 ExitCodes result = (ExitCodes)NativeMethods.CreateContext((ulong)storeSize, ref NativeContext);
61
62 switch (result)
63 {
64 case ExitCodes.EXIT_SUCCESS:
65 break;
66 case ExitCodes.ERR_CANNOT_CREATE_CONTEXT:
67 throw new MuPDFException("Cannot create MuPDF context", result);
68 case ExitCodes.ERR_CANNOT_REGISTER_HANDLERS:
69 throw new MuPDFException("Cannot register document handlers", result);
70 default:
71 throw new MuPDFException("Unknown error", result);
72 }
73 }
74
75 /// <summary>
76 /// Wrap an existing pointer to a native MuPDF context object.
77 /// </summary>
78 /// <param name="nativeContext">The pointer to the native context that should be used.</param>
79 internal MuPDFContext(IntPtr nativeContext)
80 {
81 this.NativeContext = nativeContext;
82 }
83
84 /// <summary>
85 /// Evict all items from the resource cache store (freeing the memory where they were held).
86 /// </summary>
87 public void ClearStore()
88 {
89 NativeMethods.EmptyStore(this.NativeContext);
90 }
91
92 /// <summary>
93 /// Evict items from the resource cache store (freeing the memory where they were held) until the the size of the store drops to the specified fraction of the current size.
94 /// </summary>
95 /// <param name="fraction">The fraction of the current size that constitutes the target size of the store. If this is &lt;= 0, the cache is cleared. If this is &gt;= 1, nothing happens.</param>
96 public void ShrinkStore(double fraction)
97 {
98 if (fraction <= 0)
99 {
100 ClearStore();
101 }
102 else if (Math.Round(fraction * 100) < 100)
103 {
104 NativeMethods.ShrinkStore(this.NativeContext, (uint)Math.Round(fraction * 100));
105 }
106 }
107
108 private bool disposedValue;
109
110 ///<inheritdoc/>
111 protected virtual void Dispose(bool disposing)
112 {
113 if (!disposedValue)
114 {
115 if (disposing)
116 {
117 // TODO: eliminare lo stato gestito (oggetti gestiti)
118 }
119
120 NativeMethods.DisposeContext(NativeContext);
121 disposedValue = true;
122 }
123 }
124
125 ///<inheritdoc/>
126 ~MuPDFContext()
127 {
128 Dispose(disposing: false);
129 }
130
131 ///<inheritdoc/>
132 public void Dispose()
133 {
134 Dispose(disposing: true);
135 GC.SuppressFinalize(this);
136 }
137 }
