166  /// <summary>
167  /// Gets the address of the character that contains the specified <paramref name="point"/> in page units.
168  /// </summary>
169  /// <param name="point">The point that must be closest to the character. This is expressed in page units (i.e. with a zoom factor of 1).</param>
170  /// <param name="includeImages">If this is <see langword="true"/>, blocks containing images may be returned. Otherwise, only blocks containing text are considered.</param>
171  /// <returns>The address of the character closest to the specified <paramref name="point"/> This is <see langword="null"/> only if the page contains no characters.</returns>
172  public MuPDFStructuredTextAddress? GetClosestHitAddress(PointF point, bool includeImages)
173  {
174  float minDistance = float.MaxValue;
175  MuPDFStructuredTextAddress? closestHit = null;
176 
177  float minBlockDistance = float.MaxValue;
178  float minLineDistance = float.MaxValue;
179 
180  for (int i = 0; i < this.Count; i++)
181  {
182  if (includeImages || this[i].Type == MuPDFStructuredTextBlock.Types.Text)
183  {
184  float dx = Math.Max(0, Math.Max(this[i].BoundingBox.X0 - point.X, point.X - this[i].BoundingBox.X1));
185  float dy = Math.Max(0, Math.Max(this[i].BoundingBox.Y0 - point.Y, point.Y - this[i].BoundingBox.Y1));
186  float blockDist = dx * dx + dy * dy;
187 
188  if (this[i].BoundingBox.Contains(point) || blockDist < minBlockDistance)
189  {
190  if (blockDist < minBlockDistance)
191  {
192  minBlockDistance = blockDist;
193  minLineDistance = float.MaxValue;
194  }
195 
196  for (int j = 0; j < this[i].Count; j++)
197  {
198  dx = Math.Max(0, Math.Max(this[i][j].BoundingBox.X0 - point.X, point.X - this[i][j].BoundingBox.X1));
199  dy = Math.Max(0, Math.Max(this[i][j].BoundingBox.Y0 - point.Y, point.Y - this[i][j].BoundingBox.Y1));
200  float lineDist = dx * dx + dy * dy;
201 
202  if (this[i][j].BoundingBox.Contains(point) || lineDist < minLineDistance)
203  {
204  if (lineDist < minLineDistance)
205  {
206  minLineDistance = lineDist;
207  }
208 
209  for (int k = 0; k < this[i][j].Count; k++)
210  {
211  if (this[i][j][k].BoundingQuad.Contains(point))
212  {
213  return new MuPDFStructuredTextAddress(i, j, k);
214  }
215  else
216  {
217  //The quads should be small enough that the error due to only checking vertices and not sides is negligible. Also, since the square root is monotonous, we can skip it.
218  float minDist = (point.X - this[i][j][k].BoundingQuad.UpperLeft.X) * (point.X - this[i][j][k].BoundingQuad.UpperLeft.X) + (point.Y - this[i][j][k].BoundingQuad.UpperLeft.Y) * (point.Y - this[i][j][k].BoundingQuad.UpperLeft.Y);
219  minDist = Math.Min(minDist, (point.X - this[i][j][k].BoundingQuad.UpperRight.X) * (point.X - this[i][j][k].BoundingQuad.UpperRight.X) + (point.Y - this[i][j][k].BoundingQuad.UpperRight.Y) * (point.Y - this[i][j][k].BoundingQuad.UpperRight.Y));
220  minDist = Math.Min(minDist, (point.X - this[i][j][k].BoundingQuad.LowerRight.X) * (point.X - this[i][j][k].BoundingQuad.LowerRight.X) + (point.Y - this[i][j][k].BoundingQuad.LowerRight.Y) * (point.Y - this[i][j][k].BoundingQuad.LowerRight.Y));
221  minDist = Math.Min(minDist, (point.X - this[i][j][k].BoundingQuad.LowerLeft.X) * (point.X - this[i][j][k].BoundingQuad.LowerLeft.X) + (point.Y - this[i][j][k].BoundingQuad.LowerLeft.Y) * (point.Y - this[i][j][k].BoundingQuad.LowerLeft.Y));
222 
223  if (minDist < minDistance)
224  {
225  minDistance = minDist;
226  closestHit = new MuPDFStructuredTextAddress(i, j, k);
227  }
228  }
229  }
230  }
231  }
232  }
233  }
234  }
235 
236  return closestHit;
237  }

456  /// <summary>
457  /// Searches for the specified <see cref="Regex"/> in the text of the page. A single match cannot span multiple lines.
458  /// </summary>
459  /// <param name="needle">The <see cref="Regex"/> to search for.</param>
460  /// <returns>A lazy collection of <see cref="MuPDFStructuredTextAddressSpan"/>s representing all the occurrences of the <paramref name="needle"/> in the text.</returns>
461  public IEnumerable<MuPDFStructuredTextAddressSpan> Search(Regex needle)
462  {
463 
464  for (int i = 0; i < this.Count; i++)
465  {
466  if (this[i].Type == MuPDFStructuredTextBlock.Types.Text)
467  {
468  for (int j = 0; j < this[i].Count; j++)
469  {
470  foreach (Match match in needle.Matches(this[i][j].Text))
471  {
472  if (match.Success)
473  {
474  int startIndex = 0;
475  int kStart = 0;
476 
477  while (startIndex < match.Index)
478  {
479  startIndex += this[i][j][kStart].Character.Length;
480  kStart++;
481  }
482 
483  if (startIndex > match.Index)
484  {
485  kStart--;
486  }
487 
488  int length = 0;
489  int kEnd = kStart - 1;
490 
491  while (length < match.Length)
492  {
493  kEnd++;
494  length += this[i][j][kEnd].Character.Length;
495  }
496 
497  yield return new MuPDFStructuredTextAddressSpan(new MuPDFStructuredTextAddress(i, j, kStart), new MuPDFStructuredTextAddress(i, j, kEnd));
498  }
499  }
500  }
501  }
